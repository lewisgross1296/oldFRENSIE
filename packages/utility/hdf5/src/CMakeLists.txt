# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_HDF5_HEADERS *.hpp)
FILE(GLOB UTILITY_HDF5_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_HDF5_HEADERS = ${UTILITY_HDF5_HEADERS}")
  MESSAGE("UTILITY_HDF5_SOURCES = ${UTILITY_HDF5_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_hdf5)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_HDF5_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${HDF5} utility_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_HDF5_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
