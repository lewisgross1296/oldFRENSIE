# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_CORE_HEADERS *.hpp)
FILE(GLOB UTILITY_CORE_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_CORE_HEADERS = ${UTILITY_CORE_HEADERS}")
  MESSAGE("UTILITY_CORE_SOURCES = ${UTILITY_CORE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_core)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_CORE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} 
  ${TEUCHOS_PARAMETER_LIST} ${TEUCHOS_NUMERICS} ${TEUCHOS_COMM} utility_units)
IF(${FRENSIE_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_CORE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
