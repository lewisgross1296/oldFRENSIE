# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_INTEGRATOR_HEADERS *.hpp)
FILE(GLOB UTILITY_INTEGRATOR_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_INTEGRATOR_HEADERS = ${UTILITY_INTEGRATOR_HEADERS}")
  MESSAGE("UTILITY_INTEGRATOR_SOURCES = ${UTILITY_INTEGRATOR_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_integrator)

# Create the utility integrator library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_INTEGRATOR_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utility_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_INTEGRATOR_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
