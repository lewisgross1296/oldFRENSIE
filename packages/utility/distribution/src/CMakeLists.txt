# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_DISTRIBUTION_HEADERS *.hpp)
FILE(GLOB UTILITY_DISTRIBUTION_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_DISTRIBUTION_HEADERS = ${UTILITY_DISTRIBUTION_HEADERS}")
  MESSAGE("UTILITY_DISTRIBUTION_SOURCES = ${UTILITY_DISTRIBUTION_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_dist)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_DISTRIBUTION_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utility_core utility_prng)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_DISTRIBUTION_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
