# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB MONTE_CARLO_SOURCE_NATIVE_HEADERS *.hpp)
FILE(GLOB MONTE_CARLO_SOURCE_NATIVE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("MONTE_CARLO_SOURCE_NATIVE_HEADERS = ${MONTE_CARLO_SOURCE_NATIVE_HEADERS}")
  MESSAGE("MONTE_CARLO_SOURCE_NATIVE_SOURCES = ${MONTE_CARLO_SOURCE_NATIVE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME monte_carlo_source_native)

# Create the monte_carlo_source_native library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${MONTE_CARLO_SOURCE_NATIVE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} utility_prng utility_dist monte_carlo_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${MONTE_CARLO_SOURCE_NATIVE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
