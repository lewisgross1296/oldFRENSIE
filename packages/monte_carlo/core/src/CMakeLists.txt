# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB MONTE_CARLO_CORE_HEADERS *.hpp)
FILE(GLOB MONTE_CARLO_CORE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("MONTE_CARLO_CORE_HEADERS = ${MONTE_CARLO_CORE_HEADERS}")
  MESSAGE("MONTE_CARLO_CORE_SOURCES = ${MONTE_CARLO_CORE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME monte_carlo_core)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${MONTE_CARLO_CORE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} ${TEUCHOS_PARAMETER_LIST} ${Boost_LIBRARIES} utility_core utility_hdf5 geometry_core)

IF(${FRENSIE_ENABLE_DAGMC})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_dagmc)
ENDIF()

IF(${FRENSIE_ENABLE_ROOT})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_root)
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${MONTE_CARLO_CORE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
