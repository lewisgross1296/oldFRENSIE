# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB MONTE_CARLO_ESTIMATOR_NATIVE_HEADERS *.hpp)
FILE(GLOB MONTE_CARLO_ESTIMATOR_NATIVE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("MONTE_CARLO_ESTIMATOR_NATIVE_HEADERS = ${MONTE_CARLO_ESTIMATOR_NATIVE_HEADERS}")
  MESSAGE("MONTE_CARLO_ESTIMATOR_NATIVE_SOURCES = ${MONTE_CARLO_ESTIMATOR_NATIVE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME monte_carlo_estimator_native)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${MONTE_CARLO_ESTIMATOR_NATIVE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} ${MOAB} ${Boost_LIBRARIES} utility_dist monte_carlo_core)

IF(FRENSIE_ENABLE_DAGMC)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_dagmc)
ENDIF()
IF(FRENSIE_ENABLE_ROOT)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_root)
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${MONTE_CARLO_ESTIMATOR_NATIVE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
