# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB GEOMETRY_DAGMC_HEADERS *.hpp)
FILE(GLOB GEOMETRY_DAGMC_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("GEOMETRY_DAGMC_HEADERS = ${GEOMETRY_DAGMC_HEADERS}")
  MESSAGE("GEOMETRY_DAGMC_SOURCES = ${GEOMETRY_DAGMC_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME geometry_dagmc)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${GEOMETRY_DAGMC_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} ${TEUCHOS_PARAMETER_LIST} ${MOAB} utility_core geometry_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${GEOMETRY_DAGMC_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
