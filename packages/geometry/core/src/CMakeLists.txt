# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB GEOMETRY_CORE_HEADERS *.hpp)
FILE(GLOB GEOMETRY_CORE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("GEOMETRY_CORE_HEADERS = ${GEOMETRY_CORE_HEADERS}")
  MESSAGE("GEOMETRY_CORE_SOURCES = ${GEOMETRY_CORE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME geometry_core)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${GEOMETRY_CORE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} utility_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${GEOMETRY_CORE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
