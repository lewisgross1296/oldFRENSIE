# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB GEOMETRY_ROOT_HEADERS *.hpp)
FILE(GLOB GEOMETRY_ROOT_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("GEOMETRY_ROOT_HEADERS = ${GEOMETRY_ROOT_HEADERS}")
  MESSAGE("GEOMETRY_ROOT_SOURCES = ${GEOMETRY_ROOT_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME geometry_root)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${GEOMETRY_ROOT_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${ROOT} ${TEUCHOS_CORE} ${TEUCHOS_COMM} ${TEUCHOS_PARAMETER_LIST} utility_core geometry_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${GEOMETRY_ROOT_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
