# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB GEOMETRY_NATIVE_HEADERS *.hpp)
FILE(GLOB GEOMETRY_NATIVE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("GEOMETRY_NATIVE_HEADERS = ${GEOMETRY_NATIVE_HEADERS}")
  MESSAGE("GEOMETRY_NATIVE_SOURCES = ${GEOMETRY_NATIVE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME geometry_native)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${GEOMETRY_NATIVE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_NUMERICS} ${TEUCHOS_PARAMETER_LIST} utility_core utility_prng geometry_core)
IF(${FRENSIE_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${GEOMETRY_NATIVE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
